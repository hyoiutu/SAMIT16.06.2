\documentclass[a4j]{jsarticle}
\usepackage{ifthen}
\makeatletter

\newcount\m@mod
\newcount\n@mod
\newcount\result@mod

\newcount\m@GCD
\newcount\n@GCD

\newcount \inputM
\newcount \inputN

\inputM = 4152
\inputN = 4125

\newcommand{\mod}[2]{%
  \m@mod=#1%
  \n@mod=#2%
  \result@mod=\m@mod%
  \divide \m@mod by \n@mod%
  \multiply \n@mod by \m@mod%
  \multiply \n@mod by -1%
  \advance \result@mod by \n@mod%
}

\newcommand{\GCD}[2]{%
  \m@GCD=#1%
  \n@GCD=#2%
  \mod{\m@GCD}{\n@GCD}%
  \ifthenelse{\result@mod=0}{\the\n@GCD}{\GCD{\n@GCD}{\result@mod}}%
}
\makeatother

\begin{document}
  \LaTeX GCD(\the\inputM ,\the\inputN)=\GCD{\inputM}{\inputN}
\end{document}
