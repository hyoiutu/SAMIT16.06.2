\read-1 to \inputN

\newcommand{\mod}[2]{%
  \m@mod=#1%
  \n@mod=#2%
  \result@mod=\m@mod%
  \divide \m@mod by \n@mod%
  \multiply \n@mod by \m@mod%
  \multiply \n@mod by -1%
  \advance \result@mod by \n@mod%
}

\newcommand{\GCD}[2]{%
  \m@GCD=#1%
  \n@GCD=#2%
  \mod{\m@GCD}{\n@GCD}%
  \ifthenelse{\result@mod=0}{\the\n@GCD}{\GCD{\n@GCD}{\result@mod}}%
}
